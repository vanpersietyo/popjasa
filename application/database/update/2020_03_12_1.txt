DROP TABLE IF EXISTS `trans_bank_in`;
CREATE TABLE `trans_bank_in`  (
  `ID_TRANS` VARCHAR(25) CHARACTER SET latin1 COLLATE latin1_swedish_ci NOT NULL COMMENT 'ID TRANSAKSI (TBO-)',
  `KD_BANK` VARCHAR(255) CHARACTER SET latin1 COLLATE latin1_swedish_ci NULL DEFAULT NULL COMMENT 'KODE BANK DARI MASTER BANK',
  `SLD_MASUK` INT(11) NULL DEFAULT NULL COMMENT 'JUMLAH SALDO YANG MASUK',
  `TGL_BUAT` DATETIME(0) NULL DEFAULT NULL COMMENT 'TGL BUAT',
  `ID_OPR` VARCHAR(255) CHARACTER SET latin1 COLLATE latin1_swedish_ci NULL DEFAULT NULL COMMENT 'ID OPERATOR',
  `ST_DATA` VARCHAR(2) CHARACTER SET latin1 COLLATE latin1_swedish_ci NULL DEFAULT '0' COMMENT '1=KONFIRMASI 0=BLM KONFIRMASI',
  `TGL_TRANS` DATETIME(0) NULL DEFAULT NULL,
  `KETERANGAN` VARCHAR(255) CHARACTER SET latin1 COLLATE latin1_swedish_ci NULL DEFAULT NULL,
  PRIMARY KEY (`ID_TRANS`) USING BTREE
) ENGINE = MYISAM CHARACTER SET = latin1 COLLATE = latin1_swedish_ci COMMENT = 'transasksi bank masuk' ROW_FORMAT = DYNAMIC;
